# HTTP redirect
server {
    listen      80;
    server_name dev-erp.apptimate.lk;
    include     snippets/letsencrypt.conf;

    location / {
        return 301 https://dev-erp.apptimate.lk$request_uri;
    }
}
server {
    listen                  443 ssl http2;
    server_name             dev-erp.apptimate.lk;
    set                     $base /var/www/apptimate;
    root                    $base/apptimate-web-v3.0/public;

    # SSL
    ssl_certificate         /etc/letsencrypt/live/dev-erp.apptimate.lk/fullchain.pem;
    ssl_certificate_key     /etc/letsencrypt/live/dev-erp.apptimate.lk/privkey.pem;
    ssl_trusted_certificate /etc/letsencrypt/live/dev-erp.apptimate.lk/chain.pem;

    # security
    include                 snippets/security.conf;

    # logging
    access_log              off;
    error_log               /var/log/nginx/error.log crit;

    # index.php
    index                   index.php;

    # index.php fallback
    location / {
        try_files $uri $uri/ /index.php?$query_string;
    }

    # additional config
    include snippets/general.conf;

    # handle .php
    location ~ \.php$ {
        fastcgi_pass unix:/var/run/php/php7.4-fpm.sock;
        include      snippets/php_fastcgi.conf;
    }
}
